let random ~m ~n =
  ones ~m ~n |> Array.map (fun rows ->
      Array.map (fun _v -> Random.float 1.0) rows)

let add m1 m2 =
  if (size_equals m1 m2) then
    let mat = zeros ~m:(height m1) ~n:(width m1) in
    for i = 0 to (height mat) - 1 do
      mat.(i) <- Array.map2 (+.) m1.(i) m2.(i)
    done;
    Some mat
  else
    None

let sub m1 m2 =
  if (size_equals m1 m2) then
    let m = zeros ~m:(height m1) ~n:(width m1) in
    for i=0 to (height m) - 1 do
      m.(i) <- Array.map2 (-.) m1.(i) m2.(i)
    done;
    Some m
  else
    None

let scalar_mult m scale =
  m |> Array.map (fun rows ->
      Array.map (fun v -> v *. scale) rows)

let transpose m = 
  let nm = zeros ~m: (width m) ~n: (height m) in
  for i = 0 to (height m) - 1 do 
    for j = i to (width m) - 1 do 
      nm.(i).(j) <- m.(j).(i);
      nm.(j).(i) <- m.(i).(j);
    done
  done;
  nm
